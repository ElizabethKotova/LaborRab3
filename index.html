Функция min()

// Новая реализация функции min()
function min(a, b) {
    if (isNaN(a) || isNaN(b)) {
        alert("Пожалуйста, введите числа");
        return null;
    }
    return a < b ? a : b;
}
// Функция для интерактивного тестирования
function testMinFunction() {
    const num1 = parseFloat(prompt("Введите первое число:"));
    if (isNaN(num1)) {
        alert("Это не число!");
        return;
    }
    const num2 = parseFloat(prompt("Введите второе число:"));
    if (isNaN(num2)) {
        alert("Это не число!");
        return;
    }
    const result = min(num1, num2);
    alert(`Минимальное число: ${result}`);
}

Подтверждение возраста

function checkAge() {
    let age;
    let confirmed = false;
    while (!confirmed) {
        age = prompt("Сколько вам лет?");
        if (age === null) return;
        confirmed = confirm(`Вы уверены, что вам ${age} лет?`);
        if (!confirmed) alert("Давайте попробуем ещё раз!");
    }
    alert(`Спасибо, ваш возраст: ${age} лет`);
}
